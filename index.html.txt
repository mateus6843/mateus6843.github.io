    <script>
        // JavaScript para a lógica do Quiz
        const questionContainer = document.getElementById('question-container');
        const quizContent = document.getElementById('quiz-content');
        const resultContainer = document.getElementById('result-container');
        const submitButton = document.getElementById('submit-btn');

        // Lista de perguntas e respostas
        const questions = [
            {
                question: "Qual sua ideia de um encontro perfeito?",
                answers: ["Jantar romântico à luz de velas", "Cinema e pipoca, bem agarradinho", "Aventura na natureza", "Festa com os amigos até o sol raiar"]
            },
            {
                question: "Se o amor fosse um sabor, qual seria?",
                answers: ["Doce como chocolate", "Azedinho como maracujá", "Quente como pimenta", "Refrescante como hortelã"]
            },
            {
                question: "O que mais te atrai em alguém?",
                answers: ["O sorriso", "O olhar misterioso", "O bom humor", "A inteligência"]
            },
            {
                question: "Você me daria a bct, ein safada?",
                answers: ["Com certeza, bb!", "Só se pedir com jeitinho...", "Talvez, se você merecer", "Vem descobrir!"]
            },
            {
                question: "Para selar o destino, qual a sua vibe de casal?",
                answers: ["Casal que viaja o mundo", "Casal que maratona séries", "Casal fitness", "Casal que constrói um império"]
            }
        ];

        function startQuiz() {
            renderAllQuestions();
        }

        function renderAllQuestions() {
            questionContainer.innerHTML = ''; // Limpa o container

            questions.forEach((currentQuestion, index) => {
                // Cria um container para cada bloco de pergunta
                const questionBlock = document.createElement('div');
                questionBlock.className = 'question-block';

                // Cria o texto da pergunta
                const questionText = document.createElement('p');
                questionText.className = 'question-text';
                questionText.textContent = `${index + 1}. ${currentQuestion.question}`;

                // Cria o container para as respostas
                const answersDiv = document.createElement('div');
                answersDiv.className = 'answers';

                // Cria os botões de resposta
                currentQuestion.answers.forEach(answer => {
                    const button = document.createElement('button');
                    button.className = 'btn';
                    button.textContent = answer;
                    button.onclick = (event) => {
                        // Remove a seleção de outros botões na mesma pergunta
                        const siblings = event.target.parentElement.children;
                        for (let sib of siblings) {
                            sib.classList.remove('selected');
                        }
                        // Adiciona a classe 'selected' ao botão clicado
                        event.target.classList.add('selected');
                    };
                    answersDiv.appendChild(button);
                });

                questionBlock.appendChild(questionText);
                questionBlock.appendChild(answersDiv);
                questionContainer.appendChild(questionBlock);
            });
        }

        function showResult() {
            quizContent.classList.add('fade-out');

            setTimeout(() => {
                quizContent.style.display = 'none';
                resultContainer.style.display = 'block';
                resultContainer.classList.add('fade-in-new');
            }, 400); // Duração da animação de saída
        }
        
        // Adiciona o evento de clique ao botão de submissão
        submitButton.addEventListener('click', showResult);

        // Inicia o quiz assim que a página é carregada
        window.onload = startQuiz;
    </script>
